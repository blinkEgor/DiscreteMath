// Алгоритм выделения графа с минимальным общим весом, связывающего все вершины в данном связном взвешенном графе

begin
    v:= произвольная вершина;
    u:= ближайшая соседняя вершина;
    связать v и u;
    while остаются неприсоединённые вершины do
        begin
            u:= неприсоединённая вершина, ближайшая
                к одной из присоединённых вершин;
            соединить u с ближжайшей из присоединённых вершинж
        end
end
